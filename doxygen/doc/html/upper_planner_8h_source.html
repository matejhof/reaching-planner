95  void sendTrajectory()
96  {
97  Bottle &outTraj = prepare();
98  outTraj.clear();
99  Bottle traj, viaPoint;
100  traj.clear();
101  viaPoint.clear();
102 
103  ts.update();
104 
105  traj.addString(controlPoint); // Added on 21/03/2016 for sending trajectory of different control points
106 
107  traj.addInt(trajectory.size());
108  traj.addInt(nDim);
109  for (int j=0; j<trajectory.size(); j++)
110  {
111  viaPoint.clear();
112  for (int i=0; i<nDim; i++)
113  viaPoint.addDouble(trajectory[j][i]);
114  traj.append(viaPoint);
115  }
116  outTraj.addList().read(traj);
117  setEnvelope(ts);
118  write();
119  }
120 };
