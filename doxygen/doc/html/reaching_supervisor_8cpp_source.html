121  indexCurSegment=0;

123 
124  tempWaypoint->stop();

130 
131 
132  deque<waypointTrajectory> &listTraject = planPortIn.getListTrajectory();

135 
136  printf("Got some thing. listTrajectory.size()= %d\n",(int) listTraject.size());
137 
138  for (int i=0; i<listTraject.size(); i++)
139  {
140 
141  printf("numberWaypoint = %d\n",numberWaypoint);
142  string tempCtrlPtName = listTraject[i].getCtrlPointName();
143  printf("CtrlPointName = %s\n",tempCtrlPtName.c_str());
144  vector<Vector> tempTrajectory = listTraject[i].getWaypoints();

148  }

151 
152  listTrajectories = planPortIn.getListTrajectory();
153  }
154 
155  if (listTrajectories.size()>0)
156  {
157  // For each trajectory of each controlled point:
158  // 1 Get 2 sucessive waypoints of trajectory and put to Segment
159  // 2 Generate temporal waypoints between them
160  // 3 Integrate after each period
161  // 4 Check new particle is close to the 2nd waypoint
162  // a YES:
163  // - Set 2nd waypoint and next waypoint in trajectory as Segment
164  // - Come back to 1
165  // b NO: come back to 2 and continue
166  //
167 
168  ctrlPointsNames.clear();
169  for (int i=0; i<listTrajectories.size(); i++)
170  {
171  string tempCtrlPtName = listTrajectories[i].getCtrlPointName();
172  ctrlPointsNames.push_back(tempCtrlPtName);
173  }
174  tempWaypoint->setCtrlPointsNames(ctrlPointsNames);
175 
176  numberWaypoint = listTrajectories[0].getNbWaypoint();
177  if (indexCurSegment<numberWaypoint-1)
178  {
179  printf("indexCurSegment =%d\n", indexCurSegment);
180  vector<Vector> tempTrajectoryEE = listTrajectories[0].getWaypoints();
181  vector<Vector> tempTrajectoryEB = listTrajectories[1].getWaypoints();
182 
183  Vector x_0EE(nDim,0.0), x_0EB(nDim,0.0), x_dEE(nDim,0.0), x_dEB(nDim,0.0);
184  Vector velEE(nDim,0.0), velEB(nDim,0.0);
185 
186  vector<Vector> x_0, vel;
187  vector<Vector> x_d;
188 
189  x_0EE = tempTrajectoryEE[indexCurSegment];
190  x_dEE = tempTrajectoryEE[indexCurSegment+1];
191 
192  x_0EB = tempTrajectoryEB[indexCurSegment];
193  x_dEB = tempTrajectoryEB[indexCurSegment+1];
194 
195  speedEB = computeOtherCtrlPtSpeed(speedEE,x_0EE,x_dEE,x_0EB,x_dEB);
196 
197  velEE = computeVelFromSegment(speedEE,x_0EE,x_dEE);
198  velEB = computeVelFromSegment(speedEB,x_0EB,x_dEB);
199 
200  //multi-waypoints
201  x_0.push_back(x_0EE);
202  x_0.push_back(x_0EB);
203  vel.push_back(velEE);
204  vel.push_back(velEB);
205  x_d.push_back(x_dEE);
206  x_d.push_back(x_dEB);
207 
208 
209  printf("finishedCurSegment =%d \n",finishedCurSegment);
210  if (finishedCurSegment)
211  {
212  printf("check finishedCurSegment 1\n");
213  tempWaypoint->stop();
214 
215  printf("check finishedCurSegment 2\n");

218 
219  printf("check finishedCurSegment 3\n");
220  tempWaypoint->setLastWaypoint(x_d);
221 
222  printf("check finishedCurSegment 4\n");
223  tempWaypoint->start();
224 
225  printf("check finishedCurSegment 5\n");
226  }
227 
228  vector<Vector> x_n = tempWaypoint->getParticle();
229  printf("check getParticle()\n");
230 
231  printf("norm(x_nEE-x_dEE) = %f\t norm(x_nEB-x_dEB) = %f\n", norm(x_n[0]-x_d[0]), norm(x_n[1]-x_d[1]));
232 
233  if ((tempWaypoint->checkFinished()))
234  {
235  finishedCurSegment = true;
236  printf("reaching waypoint %d-th & finish current segment \n",indexCurSegment+1);

238  }

250  }
251 
252

282 double reachingSupervisor::distWpWp(const Vector &wp1, const Vector &wp2)
283 {
284  double distance = 1000;
285  if ((wp1.size()==nDim) && (wp2.size()==nDim))
286  {
287  double sqSum = 0;
288  for (int i=0; i<nDim; i++)
289  sqSum += pow((wp1[i]-wp2[i]),2.0);
290  distance = sqrt(sqSum);
291  }
292  return distance;
293 }
