160 void singlePlanner::updatePlanner()
161 {
162  rrts.initialize ();
163 
164  // This parameter should be larger than 1.5 for asymptotic

171 vector<Vector> singlePlanner::generateTrajectory()
172 //void singlePlanner::generateTrajectory(vector<Vector> newBest)
173 {
174  Vector vertexBest(nDim,0.0);
175  Vector vertexBestRoot(nDim,0.0);
176  list<double*> traject;
177 
178 
179  clock_t start = clock();
180  //Time startTime = Time::now();
181  //*******************************************************************//
182  // Main loop to generate a plan
183  for (int i = 0; i < nIter; i++)
184  {
185  rrts.iteration ();
186  clock_t finish = clock();
187  //Time finishTime = Time::now();
188  solveTime = ((double)(finish-start))/CLOCKS_PER_SEC;
189  if (solveTime>=deadline)
190  break;
191  // end of added part
192  }
193  //*******************************************************************//
194 
195  rrts.getBestTrajectory(traject);
196  for (list<double*>::const_iterator itt = traject.begin(); itt!=traject.end();itt++)
197  {
198  double* stateRef = *itt;
199  for (int i=0; i<nDim; i++)
200  {
201  vertexBest[i] = stateRef[i];
202  }
203  bestTraj.push_back(vertexBest);
204  convertPosFromSimToRootFoR(vertexBest,vertexBestRoot);
205  bestTrajRoot.push_back(vertexBestRoot);
206  }
207 
208  cout<<"\nTrajectory is generated in "<<solveTime<<" seconds !!!"<< endl;
209  cout<<endl;
210  return bestTraj;
211 }
