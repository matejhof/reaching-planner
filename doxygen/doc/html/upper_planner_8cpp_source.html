1106  const Vector &pointj, const double &lengthLimb,
1107  Vector &newPoint)
1108 {

1110  Vector Xt1(nDim,0.0), Xt2(nDim,0.0);

1113  double eps = 0.001;
1114 
1115  bool result = false;
1116 
1117  for (int i=0; i<nDim; i++)
1118  {
1119  v[i]=point2[i]-point1[i];
1120  a[i]=pointj[i]-point1[i];
1121  }
1122  coefA = pow(v[0],2.0) + pow(v[1],2.0) + pow(v[2],2.0);
1123  coefB = -2*(a[0]*v[0] + a[1]*v[1] + a[2]*v[2]);
1124  coefC = pow(a[0],2.0) + pow(a[1],2.0) + pow(a[2],2.0) - pow(lengthLimb,2.0);
1125 
1126  delta = pow(coefB,2.0)-4*coefA*coefC;
1127 
1128  if (delta>=0)
1129  {
1130  t1 = (-coefB+sqrt(delta))/(2*coefA);
1131  t2 = (-coefB-sqrt(delta))/(2*coefA);
1132  for (int i=0; i<nDim; i++)
1133  {
1134  Xt1[i] = point1[i] + v[i]*t1;
1135  Xt2[i] = point1[i] + v[i]*t2;
1136  }
1137  double dist_Xt1_point1 = distWpWp(Xt1,point1);
1138  double dist_Xt1_point2 = distWpWp(Xt1,point2);
1139  double dist_Xt2_point1 = distWpWp(Xt2,point1);
1140  double dist_Xt2_point2 = distWpWp(Xt2,point2);
1141  double dist_point1_point2 = distWpWp(point1,point2);
1142 
1143  if((abs(10*(dist_Xt1_point1+dist_Xt1_point2-dist_point1_point2))/10.0)<=eps)
1144  {
1145  newPoint = Xt1;
1146  result = true;
1147  }
1148 
1149  if((abs(10*(dist_Xt2_point1+dist_Xt2_point2-dist_point1_point2))/10.0)<=eps)
